import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.mockito.Mockito.mock;
import static org.mockito.Mockito.when;
import static org.opensearch.sql.ast.tree.Sort.NullOrder.NULL_FIRST;
import static org.opensearch.sql.ast.tree.Sort.NullOrder.NULL_LAST;

import org.opensearch.sql.sql.antlr.parser.OpenSearchSQLParser.OrderByElementContext;

@DisplayNameGeneration(DisplayNameGenerator.ReplaceUnderscores.class)
class ParserUtilsTest {

  @Test
  void createSortOption_implicitAsc_noNulls_defaultsNullFirst() {
    OrderByElementContext ctx = mock(OrderByElementContext.class);
    ctx.order = null;
    when(ctx.FIRST()).thenReturn(null);
    when(ctx.LAST()).thenReturn(null);

    SortOption opt = ParserUtils.createSortOption(ctx);
    assertEquals(NULL_FIRST, opt.getNullOrder());
  }

  @Test
  void createSortOption_explicitDesc_noNulls_defaultsNullLast() {
    OrderByElementContext ctx = mock(OrderByElementContext.class);
    Token descToken = mock(Token.class);
    when(descToken.getText()).thenReturn("DESC");
    ctx.order = descToken;
    when(ctx.FIRST()).thenReturn(null);
    when(ctx.LAST()).thenReturn(null);

    SortOption opt = ParserUtils.createSortOption(ctx);
    assertEquals(DESC, opt.getSortOrder());
    assertEquals(NULL_LAST, opt.getNullOrder());
  }

  @Test
  void createSortOption_explicitAsc_nullsLast_honorsExplicit() {
    OrderByElementContext ctx = mock(OrderByElementContext.class);
    Token ascToken = mock(Token.class);
    when(ascToken.getText()).thenReturn("ASC");
    ctx.order = ascToken;
    when(ctx.FIRST()).thenReturn(null);
    when(ctx.LAST()).thenReturn(mock(TerminalNode.class));

    SortOption opt = ParserUtils.createSortOption(ctx);
    assertEquals(NULL_LAST, opt.getNullOrder());
  }

  @Test
  void createSortOption_explicitDesc_nullsFirst_honorsExplicit() {
    OrderByElementContext ctx = mock(OrderByElementContext.class);
    Token descToken = mock(Token.class);
    when(descToken.getText()).thenReturn("DESC");
    ctx.order = descToken;
    when(ctx.FIRST()).thenReturn(mock(TerminalNode.class));
    when(ctx.LAST()).thenReturn(null);

    SortOption opt = ParserUtils.createSortOption(ctx);
    assertEquals(NULL_FIRST, opt.getNullOrder());
  }
}
